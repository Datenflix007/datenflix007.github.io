
algorithm MaxHeapify(A, i)
  l = left(i)
  r = right(i)
  if l <= heap_size[A] and A[l] > A[i] then
    largest = l
  else
    largest = i
  end if

  if r <= heap_size[A] and A[r] > A[largest] then
    largest = r
  end if

  if largest != i then
    swap(A[i], A[largest])
    MaxHeapify(A, largest)
  end if
end algorithm

algorithm BuildMaxHeap(A)
  heap_size[A] = length[A]
  for i = floor(length[A] / 2) downto 1 do
    MaxHeapify(A, i)
  end for
end algorithm

algorithm MaxHeapInsert(A, key)
  heap_size[A] = heap_size[A] + 1
  A[heap_size[A]] = -∞
  MaxHeapIncreaseKey(A, heap_size[A], key)
end algorithm

algorithm MaxHeapIncreaseKey(A, i, key)
  if key < A[i] then
    Error
  end if
  A[i] = key
  while i > 1 and A[parent(i)] < A[i] do
    swap(A[i], A[parent(i)])
    i = parent(i)
  end while
end algorithm
