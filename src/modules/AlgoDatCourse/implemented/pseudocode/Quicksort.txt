⊳ Eingabe: Array A[1..n], linker Index p, rechter Index r
⊳ Ausgabe: Array A[p..r] aufsteigend sortiert (in-place)

algorithm Quicksort(A, p, r)
  if p < r then
    q = Partition(A, p, r)
    Quicksort(A, p, q - 1)
    Quicksort(A, q + 1, r)
  end if
end algorithm

algorithm Partition(A, p, r)
  x = A[r]
  i = p - 1
  for j = p to r - 1 do
    if A[j] <= x then
      i = i + 1
      swap(A[i], A[j])
    end if
  end for
  swap(A[i + 1], A[r])
  return i + 1
end algorithm
